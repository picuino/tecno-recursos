    //
// Sensor LDR
//
#include <Picuino.h>

// Inicia Arduino
void setup() {
   pio.begin();
   Serial.begin(19200);  // Inicializa el puerto serie
}

// Bucle principal
void loop() {
   // Lee la señal producida por el sensor LDR
   int ldr = analogRead(A1);

   // Muestra la señal por pantalla
   pio.dispWrite(ldr);
   Serial.print("LDR = ");
   Serial.println(ldr);

   // Espera medio segundo
   delay(500);
}
